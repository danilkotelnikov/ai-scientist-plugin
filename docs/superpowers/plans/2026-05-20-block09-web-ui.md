// src/api/client.ts
import { getToken } from "../lib/auth";

const BASE = import.meta.env.VITE_API_BASE ?? "https://api.vedix.ai";

export async function api<T>(path: string, init?: RequestInit): Promise<T> {
  const token = getToken();
  const r = await fetch(`${BASE}${path}`, {
    ...init,
    headers: {
      "Content-Type": "application/json",
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...init?.headers,
    },
  });
  if (!r.ok) throw new Error(`${r.status} ${await r.text()}`);
  return r.json() as Promise<T>;
}
```

// src/components/ProgressStream.tsx
import { useEffect, useState } from "react";
import { getToken } from "../lib/auth";

interface Event { id: number; kind: string; ts: number; payload: any; }

export function ProgressStream({ jobId }: { jobId: string }) {
  const [events, setEvents] = useState<Event[]>([]);
  useEffect(() => {
    const token = getToken();
    const BASE = import.meta.env.VITE_API_BASE ?? "https://api.vedix.ai";
    // SSE doesn't accept Authorization headers directly; use a one-shot token endpoint
    const url = `${BASE}/v1/api/jobs/${jobId}/events?token=${token}`;
    const es = new EventSource(url);
    es.onmessage = (ev) => {
      try {
        const data = JSON.parse(ev.data);
        setEvents(prev => [...prev, { id: parseInt(ev.lastEventId || "0"), kind: ev.type, ts: Date.now()/1000, payload: data }]);
      } catch {}
    };
    return () => es.close();
  }, [jobId]);
  return (
    <div className="font-mono text-sm space-y-1 p-4 bg-gray-50 rounded max-h-96 overflow-auto">
      {events.map(e => (
        <div key={e.id} className="flex gap-4">
          <span className="text-gray-500 w-24 shrink-0">{new Date(e.ts*1000).toLocaleTimeString()}</span>
          <span className="font-semibold w-32 shrink-0">{e.kind}</span>
          <span className="text-gray-800">{JSON.stringify(e.payload)}</span>
        </div>
      ))}
    </div>
  );
}
```

// src/pages/Providers.tsx
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { listProviders, addProvider, removeProvider, setChain } from "../api/providers";
import { useState } from "react";

const ALL_PROVIDERS = [
  "anthropic","openai","google","openrouter","together","deepseek","qwen","moonshot","zhipu",
  "gigachat","yandexgpt","mistral","cohere","local",
];

export function ProvidersPage() {
  const qc = useQueryClient();
  const { data: providers = [] } = useQuery({ queryKey: ["providers"], queryFn: listProviders });
  const addMut = useMutation({ mutationFn: ({ name, key }: { name: string; key: string }) => addProvider(name, key),
                               onSuccess: () => qc.invalidateQueries({ queryKey: ["providers"] }) });
